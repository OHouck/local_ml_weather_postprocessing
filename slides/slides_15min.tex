\section{Background}

\begin{frame}{Running ML Weather Models is Cheap but Training them is Expensive}
    \begin{itemize}
        \item Instead of spending compute power predicting upper atmosphere circulation patterns, why not focus on the weather variables most relevant to people
        \item Retraining or fine-tuning even pieces of these models may be out of reach for those poor in compute.

    \end{itemize}
    \pause 
    \vspace{0.5em}
    \textbf{Goal:} % Delays the appearance of the content below this line
    Demonstrate that is is possible to meaningfully improve the accuracy of ML weather models for the most relevant weather variables using a small amount of compute and data.
\end{frame}


\begin{frame}{Research Questions and Early Results}
\begin{itemize}
    \item Is it possible to improve forecast accuracy using easily available off the shelf tools?
    \item Can NWP and ML weather models can be corrected in the same way? 
    \item What is the trade off between forecast improvement and compute cost?
    \item We train a simple neural network to correct both a data driven ML forecast (Pangu) and a traditional NWP forecast (IFS HRES) we find success predicting 10m wind speed but not so much 2m temperature. 
    \begin{itemize}
        \item 10m Wind Speed Improvements range from 5\%-20\% for lead times 1 to 7 days
        \item 2m Temp Improvements range from  -20\% - 20\%
    \end{itemize}
\end{itemize}

\vspace{0.5em}
\textbf{Key Contribution:} Apply neural network post-processing to both NWP and ML weather models, showing that it is possible to improve the accuracy of both with perhaps some caveats.
\end{frame}

\begin{frame}{Previous Papers have found Success Post-Processing NWP Forecasts with Neural Networks}
\begin{itemize}
    \item \cite{hanDeepLearningMethod2021} trained a U-Net to correct IFS Forecasts in Northern China 
    \begin{itemize}
        \item Improvements in 2m Temp RMSE go range from 30\% to 12\% for lead times 1 to 7 days
        \item Improvements in 10m Wind Speed RMSE Range from 25\% to 20\% for lead times 1 to 7 days 
    \end{itemize}
    \item \cite{araujoPostProcessingAirTemperature2022} try to improve post-processing model GRIDPP over Norway
    \begin{itemize}
        \item Improve RMSE of GRIDPP by about 1\%. Suggests ML post-processing is most useful when there is not an existing and mature post-processing model.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Start by Manually Looking at Four Regions of Interest}
\begin{itemize}
    \item Choose a handful of regions with a variety of climates and land types. 
\end{itemize}

\begin{figure}
    \centering
    \includegraphics[width=0.8\linewidth]{finetuning/climate_zones_with_patches.png}
\end{figure}
\end{frame}


\begin{frame}{Method}
\begin{itemize}
    \item Uses Pangu and IFS HRES forecasts with lead times from 1 to 7 days
    \begin{itemize}
        \item Target Pangu predictions using ERA5 and IFS HRES predictions using its t=0 initialization
    \end{itemize}
    \item Each model tries to predict a single variable for a single lead time in a particular region. 
    \item Train neural network that takes in Pangu/IFS forecast outputs and targets MSE compared to target ERA5/IFS HRES data
    \begin{itemize}
        \item Train using a single forecast per day for years 2018-2021
        \item Test using daily forecasts in 2022
        \item Test MLP architecture with 5 hidden layers, each with 512 neurons and a UNet architecture with 128 base channels
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Headline Result: Simple MLP Seems Effective at Correcting Wind Speed but not Temperature} 
\begin{itemize}
    \item Simple MLP method works across variety of regions and climates
    \item Training an MLP correction model for a 2x2 degree region takes about 2.5 minutes on an M3 max MacBook Pro
    \begin{itemize}
        \item UNet model takes about 6 minutes to train. 
    \end{itemize}
